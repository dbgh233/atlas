```sql
-- Enable WAL mode for concurrent read/write
PRAGMA journal_mode=WAL;

-- Dead Letter Queue: failed webhook payloads for investigation and replay
CREATE TABLE IF NOT EXISTS dead_letter_queue (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    event_type TEXT NOT NULL,
    payload TEXT NOT NULL,
    error_message TEXT NOT NULL,
    error_context TEXT,
    retry_count INTEGER DEFAULT 0,
    status TEXT DEFAULT 'pending',
    created_at TEXT DEFAULT (datetime('now')),
    updated_at TEXT DEFAULT (datetime('now'))
);

-- Audit Snapshots: daily audit results for trend tracking
CREATE TABLE IF NOT EXISTS audit_snapshots (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    run_date TEXT NOT NULL,
    run_type TEXT DEFAULT 'scheduled',
    total_opportunities INTEGER NOT NULL,
    total_issues INTEGER NOT NULL,
    issues_by_type TEXT NOT NULL,
    full_results TEXT NOT NULL,
    created_at TEXT DEFAULT (datetime('now'))
);

-- Interaction Log: every human interaction with Atlas
CREATE TABLE IF NOT EXISTS interaction_log (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    interaction_type TEXT NOT NULL,
    user_id TEXT NOT NULL,
    channel_id TEXT,
    opportunity_id TEXT,
    field_name TEXT,
    old_value TEXT,
    new_value TEXT,
    context TEXT,
    created_at TEXT DEFAULT (datetime('now'))
);

-- Idempotency Keys: prevent duplicate webhook processing
CREATE TABLE IF NOT EXISTS idempotency_keys (
    key TEXT PRIMARY KEY,
    event_type TEXT NOT NULL,
    processed_at TEXT DEFAULT (datetime('now')),
    result TEXT
);

CREATE INDEX IF NOT EXISTS idx_dlq_status ON dead_letter_queue(status);
CREATE INDEX IF NOT EXISTS idx_audit_date ON audit_snapshots(run_date);
CREATE INDEX IF NOT EXISTS idx_interaction_opp ON interaction_log(opportunity_id);
CREATE INDEX IF NOT EXISTS idx_idempotency_time ON idempotency_keys(processed_at);
```

Create `app/core/database.py` — Database class:
- class Database:
  - __init__(self, db_path: str) — stores path
  - async connect() -> aiosqlite.Connection — opens connection, sets row_factory = aiosqlite.Row, enables WAL mode with `PRAGMA journal_mode=WAL`
  - async run_migrations(db: aiosqlite.Connection) — static/class method:
    1. Create _migrations table if not exists
    2. Read applied migrations from _migrations
    3. Scan migrations/ directory for *.sql files sorted by name
    4. For each unapplied migration: executescript(sql), insert into _migrations, commit
    5. Log each migration applied
  - async close(db: aiosqlite.Connection) — closes connection
- get_db(request: Request) -> aiosqlite.Connection dependency function pulling from request.app.state.db
  </action>

Create `app/models/database.py` — Typed repository classes for each table:

DLQRepository(db: aiosqlite.Connection):
- async add(event_type: str, payload: str, error_message: str, error_context: str | None = None) -> int — INSERT, return id
- async get_pending(limit: int = 50) -> list[dict] — SELECT WHERE status='pending' ORDER BY created_at, limit
- async update_status(id: int, status: str) -> None — UPDATE status and updated_at WHERE id
- async get_by_id(id: int) -> dict | None — SELECT WHERE id

AuditRepository(db: aiosqlite.Connection):
- async add(run_date: str, run_type: str, total_opps: int, total_issues: int, issues_by_type: str, full_results: str) -> int — INSERT, return id
- async get_latest(limit: int = 7) -> list[dict] — SELECT ORDER BY run_date DESC LIMIT
- async get_by_date(run_date: str) -> dict | None — SELECT WHERE run_date

InteractionRepository(db: aiosqlite.Connection):
- async add(interaction_type: str, user_id: str, channel_id: str | None = None, opportunity_id: str | None = None, field_name: str | None = None, old_value: str | None = None, new_value: str | None = None, context: str | None = None) -> int — INSERT, return id
- async get_by_opportunity(opp_id: str, limit: int = 50) -> list[dict] — SELECT WHERE opportunity_id ORDER BY created_at DESC
- async get_recent(limit: int = 100) -> list[dict] — SELECT ORDER BY created_at DESC

IdempotencyRepository(db: aiosqlite.Connection):
- async exists(key: str) -> bool — SELECT count WHERE key
- async add(key: str, event_type: str, result: str = "success") -> None — INSERT OR IGNORE
- async cleanup_old(days: int = 30) -> int — DELETE WHERE processed_at < datetime('now', '-{days} days'), return deleted count

All methods use parameterized queries (? placeholders). All return dicts (via row_factory = aiosqlite.Row, then dict(row)).
